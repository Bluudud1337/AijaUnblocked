document.addEventListener("DOMContentLoaded", function(){

let topZ = 1100;

/* ================= WINDOW SYSTEM ================= */
function bringToFront(win){ topZ++; win.style.zIndex = topZ; }

function makeDraggable(win){
  if(!win) return;
  const header = win.querySelector(".window-header");
  if(!header) return;
  let offsetX = 0, offsetY = 0, dragging = false;

  header.addEventListener("mousedown", e=>{
    bringToFront(win);
    dragging = true;
    offsetX = e.clientX - win.offsetLeft;
    offsetY = e.clientY - win.offsetTop;
  });

  document.addEventListener("mousemove", e=>{
    if(!dragging) return;
    win.style.left = e.clientX - offsetX + "px";
    win.style.top  = e.clientY - offsetY + "px";
  });

  document.addEventListener("mouseup", ()=> dragging = false);
}

window.createWindow = function(title,url){
  const win = document.createElement("div");
  win.className = "window";
  win.style.top = "120px";
  win.style.left = "120px";

  win.innerHTML = `
    <div class="window-header">
      <span>${title}</span>
      <div style="display:flex; gap:6px;">
        <button onclick="this.closest('.window').remove()">X</button>
        <button onclick="toggleFullscreen(this.closest('.window'))">⛶</button>
      </div>
    </div>
    <div class="window-body">
      <iframe src="${url}" allowfullscreen></iframe>
    </div>
  `;

  document.body.appendChild(win);
  bringToFront(win);
  makeDraggable(win);
}

window.toggleFullscreen = function(win){
  if(!win.dataset.fullscreen){
    win.dataset.prev = JSON.stringify({
      width: win.style.width,
      height: win.style.height,
      top: win.style.top,
      left: win.style.left
    });
    win.style.top = "0";
    win.style.left = "0";
    win.style.width = "100vw";
    win.style.height = "100vh";
    win.dataset.fullscreen = "true";
  } else {
    const prev = JSON.parse(win.dataset.prev);
    win.style.width = prev.width;
    win.style.height = prev.height;
    win.style.top = prev.top;
    win.style.left = prev.left;
    delete win.dataset.fullscreen;
  }
}

/* ================= BACKGROUND SETTINGS ================= */
const bgWindow = document.getElementById("bgSettings");
makeDraggable(bgWindow);

window.openBgSettings = function(){ if(!bgWindow) return; bgWindow.classList.remove("hidden"); bringToFront(bgWindow); }
window.closeBgSettings = function(){ if(!bgWindow) return; bgWindow.classList.add("hidden"); }

function setBackground(src){
  document.body.style.backgroundImage = `url(${src})`;
  localStorage.setItem("osBackground", src);
}

window.setBgFromUrl = function(){
  const input = document.getElementById("bgUrlInput");
  if(input && input.value){ setBackground(input.value); }
}

const upload = document.getElementById("bgUpload");
if(upload){
  upload.addEventListener("change", function(e){
    const file = e.target.files[0];
    if(!file) return;
    const reader = new FileReader();
    reader.onload = ev => setBackground(ev.target.result);
    reader.readAsDataURL(file);
  });
}

const savedBg = localStorage.getItem("osBackground");
if(savedBg){ document.body.style.backgroundImage = `url(${savedBg})`; }

/* ================= CLOCK SYSTEM ================= */
let is24Hour = localStorage.getItem("clockFormat") === "24";

function updateClock(){
  const clockEl = document.getElementById("clock");
  const dateEl  = document.getElementById("date");
  if(!clockEl || !dateEl) return;
  const now = new Date();
  let h = now.getHours();
  let m = now.getMinutes().toString().padStart(2,"0");
  let s = now.getSeconds().toString().padStart(2,"0");
  let period = "";
  if(!is24Hour){ period = h >= 12 ? " PM" : " AM"; h = h % 12 || 12; }
  h = h.toString().padStart(2,"0");
  clockEl.textContent = `${h}:${m}:${s}${period}`;
  dateEl.textContent = now.toLocaleDateString(undefined,{ weekday:"short", month:"short", day:"numeric" });
}
window.toggleClockFormat = function(){ is24Hour = !is24Hour; localStorage.setItem("clockFormat", is24Hour ? "24":"12"); updateClock(); }
setInterval(updateClock,1000); updateClock();
